public Int get_media_estudiantes()
{
    Int i, notas

   for(i : this.num_estudiantes)
   {
    
    notas = notas + get_nota1(this.estudiante(i));
    notas = notas + get_nota2(this.estudiante(i));
    notas = notas + get_nota3(this.estudiante(i));
   }

   return notas;
}
